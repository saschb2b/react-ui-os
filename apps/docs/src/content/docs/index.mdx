```tsx
import { Desktop } from "@react-ui-os/desktop";
import { macosTheme } from "@react-ui-os/theme-macos";

const apps = [{ id: "notes", name: "Notes", content: () => <h1>Hello, desktop.</h1> }];

<Desktop apps={apps} theme={macosTheme} />;
```
